# Phase 13: Roadmap Branch Architecture

**Status:** 🔵 Planned - Ready for Execution  
**Priority:** P1 - High  
**Duration:** ~2-3 hours  
**Dependencies:** Phase 12 (Matrix-Based Approval) ✅

---

## Objective

Establish the **roadmap branch** as the authoritative source for approved concepts, implementing the three-tier branching strategy:

```
concepts/{name} → roadmap → implementations/{name}
      │              │              │
   Concept      Approved        Code
 Development     Backlog      Execution
```

Key outcomes:
- Roadmap branch acts as "approved backlog" of concepts ready for implementation
- Fully approved concepts automatically merge to roadmap
- Implementation branches start from roadmap (not develop)
- Canvas shows roadmap as the source of truth for what's approved
- Develop stays current through periodic roadmap sync

```bash
# ensure_roadmap_branch
# Creates roadmap branch if missing, returns branch name
ensure_roadmap_branch() {
  local primary=$(detect_primary_branch)
  
  # Check local
  if git rev-parse --verify roadmap >/dev/null 2>&1; then
    echo "roadmap"
    return 0
  fi
  
  # Check remote
  if git rev-parse --verify origin/roadmap >/dev/null 2>&1; then
    git checkout -b roadmap origin/roadmap 2>/dev/null
    git checkout "$primary" 2>/dev/null
    echo "roadmap"
    return 0
  fi
  
  # Create from primary
  git checkout "$primary" 2>/dev/null
  git pull origin "$primary" 2>/dev/null
  git checkout -b roadmap 2>/dev/null
  
  # Initialize manifest
  mkdir -p .planning
  echo "# Roadmap Manifest" > .planning/ROADMAP-MANIFEST.md
  echo "" >> .planning/ROADMAP-MANIFEST.md
  echo "**Branch:** roadmap" >> .planning/ROADMAP-MANIFEST.md
  echo "**Created:** $(date -u +"%Y-%m-%dT%H:%M:%SZ")" >> .planning/ROADMAP-MANIFEST.md
  
  git add .planning/ROADMAP-MANIFEST.md
  git commit -m "docs: initialize roadmap branch"
  git push -u origin roadmap 2>/dev/null || true
  
  git checkout "$primary" 2>/dev/null
  echo "roadmap"
  return 0
}

# is_on_roadmap
# Check if a concept has been merged to roadmap
is_on_roadmap() {
  local concept_slug="$1"
  
  # Check if concept directory exists on roadmap branch
  local roadmap_exists=$(git ls-tree -d roadmap --name-only 2>/dev/null | grep -c "concepts/${concept_slug}$" || echo "0")
  
  # Also check manifest
  if [ -f ".planning/ROADMAP-MANIFEST.md" ]; then
    local in_manifest=$(grep -c "| ${concept_slug} |" .planning/ROADMAP-MANIFEST.md 2>/dev/null || echo "0")
  fi
  
  if [ "$roadmap_exists" -gt 0 ] || [ "${in_manifest:-0}" -gt 0 ]; then
    return 0  # On roadmap
  fi
  
  return 1  # Not on roadmap
}
```

## Step 1: Validate Concept Status

```bash
CONCEPT_SLUG="$1"
TIMESTAMP=$(date -u +"%Y-%m-%dT%H:%M:%SZ")

# Find concept location (supports directory and legacy file format)
CONCEPT_PATH=""
FOCUS_GROUP=""

for fg_dir in .planning/focus-groups/*/; do
    if [ -d "${fg_dir}concepts/${CONCEPT_SLUG}" ]; then
        CONCEPT_PATH="${fg_dir}concepts/${CONCEPT_SLUG}"
        FOCUS_GROUP=$(basename "$fg_dir")
        CONCEPT_TYPE="directory"
        break
    elif [ -f "${fg_dir}concepts/${CONCEPT_SLUG}.md" ]; then
        CONCEPT_PATH="${fg_dir}concepts/${CONCEPT_SLUG}.md"
        FOCUS_GROUP=$(basename "$fg_dir")
        CONCEPT_TYPE="file"
        break
    fi
done

if [ -z "$CONCEPT_PATH" ]; then
    echo "❌ Concept not found: ${CONCEPT_SLUG}"
    exit 1
fi

echo "📋 Found concept: ${CONCEPT_SLUG} in ${FOCUS_GROUP}"
```

## Step 2: Verify Full Approval

```bash
# Use Phase 12 approval system
source workflows/lib/approval-system.md

# Check approval matrix completion
if [ "$CONCEPT_TYPE" = "directory" ]; then
    IMPACT_MATRIX="${CONCEPT_PATH}/impact-matrix.md"
else
    IMPACT_MATRIX=""
fi

if [ -n "$IMPACT_MATRIX" ] && [ -f "$IMPACT_MATRIX" ]; then
    # Parse approval status from impact matrix
    PENDING=$(grep -c "status: pending" "$IMPACT_MATRIX" 2>/dev/null || echo "0")
    BLOCKED=$(grep -c "status: blocked" "$IMPACT_MATRIX" 2>/dev/null || echo "0")
    REJECTED=$(grep -c "status: rejected" "$IMPACT_MATRIX" 2>/dev/null || echo "0")
    
    if [ "$REJECTED" -gt 0 ]; then
        echo "❌ Cannot merge: Concept has rejections"
        exit 1
    fi
    
    if [ "$BLOCKED" -gt 0 ]; then
        echo "❌ Cannot merge: Concept has blocked approvals"
        exit 1
    fi
    
    if [ "$PENDING" -gt 0 ]; then
        echo "❌ Cannot merge: Concept has pending approvals (${PENDING})"
        exit 1
    fi
fi

echo "✅ All approvals verified"
```

## Step 3: Check for Existing Roadmap Entry

```bash
fi
```

## Step 4: Prepare Merge

```bash

```

## Step 5: Merge Concept Branch

```bash
CONCEPT_BRANCH="concepts/${CONCEPT_SLUG}"

# Check if concept branch exists
if ! git rev-parse --verify "$CONCEPT_BRANCH" >/dev/null 2>&1; then
    if ! git rev-parse --verify "origin/${CONCEPT_BRANCH}" >/dev/null 2>&1; then
        echo "⚠️ No concept branch found, copying concept from current state"
        
        # Manual copy instead of branch merge
        git checkout "$ORIGINAL_BRANCH"
        git checkout roadmap
        
        # Copy concept directory/file
        if [ "$CONCEPT_TYPE" = "directory" ]; then
            mkdir -p ".planning/focus-groups/${FOCUS_GROUP}/concepts"
            cp -r "$ORIGINAL_BRANCH:${CONCEPT_PATH}" ".planning/focus-groups/${FOCUS_GROUP}/concepts/${CONCEPT_SLUG}"
        fi
    else
        CONCEPT_BRANCH="origin/${CONCEPT_BRANCH}"
    fi
fi

# Perform merge
echo "🔀 Merging ${CONCEPT_BRANCH} to roadmap..."

git merge "$CONCEPT_BRANCH" --no-ff -m "feat(roadmap): add ${CONCEPT_SLUG} [fully-approved]

Concept: ${CONCEPT_SLUG}
Focus Group: ${FOCUS_GROUP}
Approved: ${TIMESTAMP}

This concept has received full matrix approval and is now
ready for implementation.

Co-authored-by: WGSD Automation <wgsd@openclaw.dev>"

if [ $? -ne 0 ]; then
    echo "❌ Merge conflict detected"
    echo "   Manual resolution required"
    git merge --abort
    git checkout "$ORIGINAL_BRANCH"
    exit 1
fi

echo "✅ Merge successful"
```

## Step 6: Update Roadmap Manifest

```bash
# Add entry to manifest
MANIFEST=".planning/ROADMAP-MANIFEST.md"

# Get approval info
if [ -f "${CONCEPT_PATH}/impact-matrix.md" ]; then
    IMPACTS=$(grep -c "focus_group:" "${CONCEPT_PATH}/impact-matrix.md" 2>/dev/null || echo "1")
    PRIORITY=$(grep "priority:" "${CONCEPT_PATH}/impact-matrix.md" | head -1 | sed 's/.*priority: //' | xargs)
else
    IMPACTS="1"
    PRIORITY="P1"
fi

# Append to manifest table
if grep -q "^| Concept |" "$MANIFEST"; then
    # Find insertion point after header
    sed -i "/^|---------|/a | ${CONCEPT_SLUG} | ${TIMESTAMP%%T*} | ${IMPACTS} FGs | ${PRIORITY} |" "$MANIFEST"
fi

# Add sync history entry
echo "| ${TIMESTAMP%%T*} | Concept Merged | ${CONCEPT_SLUG} from ${FOCUS_GROUP} |" >> "$MANIFEST"

git add "$MANIFEST"
git commit --amend --no-edit
```

## Step 7: Push and Archive

```bash
# Push roadmap
git push origin roadmap

# Archive concept branch (rename, don't delete)
if git rev-parse --verify "concepts/${CONCEPT_SLUG}" >/dev/null 2>&1; then

    git branch -m "concepts/${CONCEPT_SLUG}" "$ARCHIVE_BRANCH" 2>/dev/null || true
```

```bash
## create_impl_branch (UPDATED for Phase 13)

# Create an implementation branch from ROADMAP (requires clean state)

create_impl_branch() {
  local name="$1"
  local force_from_develop="${2:-}"  # Optional: --from-develop for emergency
  
  if [ -z "$name" ]; then
    echo "ERROR: Implementation branch name required"
    return 1
  fi
  
  local branch="implementations/$name"
  
  # Check if branch already exists
  if git rev-parse --verify "$branch" >/dev/null 2>&1; then
    echo "ERROR: Implementation branch $branch already exists"
    return 1
  fi
  
  # Determine base branch
  local base_branch="roadmap"
  
  # Phase 13: Branch from roadmap by default
  if [ "$force_from_develop" = "--from-develop" ]; then
    # Emergency/hotfix mode: branch from develop
    base_branch=$(detect_primary_branch)
    echo "⚠️ EMERGENCY MODE: Branching from $base_branch instead of roadmap"
  else
    # Normal mode: ensure roadmap exists and use it
    ensure_roadmap_branch >/dev/null
    base_branch="roadmap"
  fi
  
  # Ensure clean checkout
  local clean_result=$(ensure_clean_checkout "$base_branch")
  if [ $? -ne 0 ]; then
    echo "$clean_result"
    return 1
  fi
  
  # Create branch
  git checkout -b "$branch" "$base_branch" 2>&1
  
  if [ $? -eq 0 ]; then
    echo "OK: Created implementation branch $branch from $base_branch"
    return 0
  else
    echo "ERROR: Failed to create branch $branch"
    return 1
  fi
}
```

## template_roadmap_canvas

### Function

```bash
template_roadmap_canvas() {
    local REPO_NAME="$1"
    local STUB="$2"
    
    echo "# 📋 ${REPO_NAME} Roadmap"
    echo ""
    echo "> **Approved Concepts Ready for Implementation**"
    echo ""
    echo "---"
    echo ""
    
    # Stats section
    local total_concepts=0
    local in_progress=0
    local awaiting=0
    
    # Scan roadmap manifest
    if [ -f ".planning/ROADMAP-MANIFEST.md" ]; then
        total_concepts=$(grep -E "^\| [a-z]" ".planning/ROADMAP-MANIFEST.md" 2>/dev/null | wc -l)
    fi
    
    # Count implementation status
    if [ -d ".planning/active-implementations" ]; then
        in_progress=$(ls -1 .planning/active-implementations 2>/dev/null | wc -l)
    fi
    
    awaiting=$((total_concepts - in_progress))
    if [ $awaiting -lt 0 ]; then awaiting=0; fi
    
    echo "## 📊 Overview"
    echo ""
    echo "| Metric | Count |"
    echo "|--------|-------|"
    echo "| Total on Roadmap | ${total_concepts} |"
    echo "| In Implementation | ${in_progress} |"
    echo "| Awaiting Start | ${awaiting} |"
    echo ""
    
    # Approved concepts list
    echo "## ✅ Approved Concepts"
    echo ""
    echo "| Priority | Concept | Focus Group | Approved | Status |"
    echo "|----------|---------|-------------|----------|--------|"
    
    # Parse from manifest
    if [ -f ".planning/ROADMAP-MANIFEST.md" ]; then
        grep -E "^\| [a-z]" ".planning/ROADMAP-MANIFEST.md" | while read line; do
            # Parse fields and format
            local concept=$(echo "$line" | cut -d'|' -f2 | xargs)
            local approved=$(echo "$line" | cut -d'|' -f3 | xargs)
            local impacts=$(echo "$line" | cut -d'|' -f4 | xargs)
            local priority=$(echo "$line" | cut -d'|' -f5 | xargs)
            
            # Determine status
            local status="🟡 Awaiting"
            if [ -d ".planning/active-implementations/${concept}" ]; then
                status="🔵 In Progress"
            fi
            
            # Get focus group from concept path
            local fg=""
            for fg_dir in .planning/focus-groups/*/; do
                if [ -d "${fg_dir}concepts/${concept}" ]; then
                    fg=$(basename "$fg_dir")
                    break
                fi
            done
            
            echo "| ${priority:-P1} | **${concept}** | ${fg:-unknown} | ${approved} | ${status} |"
        done
    fi
    
    echo ""
    
    # Implementation slots
    echo "## 🚀 Implementation Capacity"
    echo ""
    echo "**Active Implementations:** ${in_progress}/4"
    echo ""
    if [ $in_progress -lt 4 ]; then
        echo "✅ **Slots Available:** $((4 - in_progress)) - Ready to start new implementations"
        echo ""
        echo "\`\`\`"
        echo "/wgsd create-implementation <concept-name>"
        echo "\`\`\`"
    else
        echo "⚠️ **At Capacity** - Complete current implementations before starting new ones"
    fi
    echo ""
    
    # How concepts get here
    echo "## 🔄 How Concepts Reach Roadmap"
    echo ""
    echo "1. **Concept Creation** → concept branch created"
    echo "2. **Impact Declaration** → focus groups notified"
    echo "3. **Matrix Approval** → each FG approves independently"
    echo "4. **Full Approval** → auto-merge to roadmap branch"
    echo "5. **Implementation** → branches from roadmap"
    echo ""
    
    # Sync info
    echo "---"
    echo ""
    echo "*Branch:* \`roadmap\` | *Updated:* $(date -u +"%Y-%m-%d %H:%M UTC")"
    echo ""
    echo "*To sync develop:* \`/wgsd roadmap-sync\`"
}
```

```bash
# Merge roadmap

Roadmap Sync: ${TIMESTAMP}
Commits merged: ${ROADMAP_AHEAD}

This brings approved concepts from the roadmap branch into ${PRIMARY}.

Sync performed by: /wgsd roadmap-sync"

MERGE_RESULT=$?

if [ $MERGE_RESULT -ne 0 ]; then
    echo ""
    echo "⚠️ MERGE CONFLICT"
    echo ""
    echo "Conflicts need manual resolution:"
    git status --short | grep "^UU"
    echo ""
    echo "Options:"
    echo "  1. Resolve conflicts and commit"
    echo "  2. Abort: git merge --abort"
    echo ""
    echo "After resolving: git push origin ${PRIMARY}"
    exit 1
fi

echo "✅ Merge successful"
```

## Step 5: Push Changes

```bash
    echo "⚠️ Push failed - may need manual push"
    echo "   Run: git push origin ${PRIMARY}"
fi

# Return to original branch
git checkout "$CURRENT" 2>/dev/null || git checkout ${PRIMARY}
```

```bash
# Add to manifest if on roadmap branch

if [ -f ".planning/ROADMAP-MANIFEST.md" ]; then
```
